double CalculateGARCHVol(string symbol, int shift)
{
   int period = 20;
   double close[];
   ArraySetAsSeries(close, true);
   if(CopyClose(symbol, Timeframe, shift, period+1, close) <= 0) return 0.0;
   double returns[];
   ArrayResize(returns, period);
   for(int i = 0; i < period; i++)
      returns[i] = close[i+1] != 0 ? MathLog(close[i] / close[i+1]) : 0.0;
   return MathStandardDeviation(returns) * MathSqrt(252 * 1440 / PeriodSeconds(Timeframe));
}
