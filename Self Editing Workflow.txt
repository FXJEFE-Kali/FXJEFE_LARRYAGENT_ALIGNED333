```python
    def _safe_self_edit(self, edit_description: str, reason: str = "user selfedit command", is_skill_edit: bool = False) -> str:
        """Advanced git-integrated self-editing with PR-like branches, auto-push, and safety."""
        target = Path("agent_v2.py")
        if not target.exists():
            return "❌ Cannot locate agent_v2.py"

        current_code = target.read_text(encoding="utf-8")
        proposed_code = current_code.rstrip() + "\n\n# ── SELF-EDIT ADDITION ──\n" + edit_description

        # Show diff preview
        import difflib
        diff = difflib.unified_diff(
            current_code.splitlines(keepends=True),
            proposed_code.splitlines(keepends=True),
            fromfile="agent_v2.py (current)",
            tofile="agent_v2.py (proposed)",
            lineterm=""
        )
        diff_text = "".join(diff) or "(no visible changes)"

        print("\n" + "="*50 + " PREVIEW DIFF " + "="*50)
        print(diff_text)
        print("="*110 + "\n")

        # Confirmation
        try:
            confirm = input("Apply this self-edit? (y/N): ").strip().lower()
            if confirm not in ("y", "yes"):
                return "✖️ Self-edit cancelled by user."
        except:
            return "✖️ Self-edit cancelled."

        # Syntax validation
        try:
            import ast
            ast.parse(proposed_code)
        except SyntaxError as e:
            return f"❌ Syntax error: {e}\nEdit rejected for safety."

        # Advanced Git workflow
        try:
            repo_path = self.working_dir
            if not Path(repo_path, ".git").exists():
                print("🔧 Initializing git repository...")
                git.Repo.init(repo_path)

            repo = git.Repo(repo_path)

            # Create PR-like branch for this edit
            timestamp = datetime.now().strftime("%Y%m%d-%H%M%S")
            edit_branch = f"selfedit-{timestamp}"
            repo.git.checkout("-b", edit_branch)

            # Stash any dirty changes
            if repo.is_dirty(untracked_files=True):
                print("Stashing current changes...")
                repo.git.stash()

            # Backup
            backup = target.with_suffix(".selfedit.bak")
            shutil.copy2(target, backup)

            # Apply edit
            target.write_text(proposed_code, encoding="utf-8")

            # Commit
            repo.git.add(target)
            commit_msg = f"Self-edit: {reason}\n\nDescription: {edit_description[:200]}"
            repo.index.commit(commit_msg)

            # Auto-push if remote origin exists
            if "origin" in [r.name for r in repo.remotes]:
                print("Pushing to remote origin...")
                repo.git.push("-u", "origin", edit_branch)
                push_status = f" (pushed to origin/{edit_branch})"
            else:
                push_status = " (no remote 'origin' configured — not pushed)"

            # Restore stash if needed
            if repo.git.stash("list"):
                repo.git.stash("pop")

            self._log_improvement("self_edit", "agent_v2.py", reason + push_status)
            return (f"✅ Self-edit applied on branch **{edit_branch}** and committed.\n"
                    f"Backup: {backup.name}{push_status}\n"
                    f"Review with: /git log")
        except Exception as e:
            if 'backup' in locals() and backup.exists():
                shutil.copy2(backup, target)
            return f"❌ Self-edit failed and was rolled back: {e}"
```

```python
    def _git_status(self) -> str:
        try:
            repo = git.Repo(self.working_dir)
            return (f"Branch: {repo.active_branch.name}\n"
                    f"Dirty: {repo.is_dirty()}\n"
                    f"Last commit: {repo.head.commit.message[:80]}...")
        except Exception as e:
            return f"Git status unavailable: {e}"

    def _git_log(self, limit: int = 10) -> str:
        try:
            repo = git.Repo(self.working_dir)
            commits = list(repo.iter_commits(max_count=limit))
            lines = ["📜 Recent Self-Edits:"]
            for c in commits:
                if "Self-edit" in c.message:
                    lines.append(f"  {c.committed_datetime.strftime('%Y-%m-%d %H:%M')} | {c.message.splitlines()[0]}")
            return "\n".join(lines) if len(lines) > 1 else "No self-edits in recent history."
        except Exception as e:
            return f"Git log unavailable: {e}"

    def _selfedit_undo(self) -> str:
        try:
            repo = git.Repo(self.working_dir)
            repo.git.reset("--hard", "HEAD~1")
            return "✅ Last self-edit undone."
        except Exception as e:
            return f"❌ Undo failed: {e}"
```
